7    abstract public class Unit {
8        /* 1.2 - доступ к свойству из текущего класса + дочерних классов */
9        protected int healthScore;
10       private int maxHealthScore; //1.7 - переменная для максимального здоровья!
11       // 1.3 - конструктор
12       public Unit(int healthScore){
13           if (healthScore < 1) {
14               throw new IllegalArgumentException("Здоровье дб положительным");
15           }
16           this.healthScore = healthScore;  // 1.4 - для текущего объекта - this.healthScore значение healthScore должно быть равно значению переданого в конструктора
17           maxHealthScore = healthScore; //1.8 - органичение по максимальному здоровью!
18       }
19       // 1.11 - геттор -> BattleUnit
20       public int getHealthScore(){
21           return healthScore;
22       }
23   
24       // 1.9 - увеличить здоровье на healthScore, но не больше изначального, если персонаж жив
25       public void plusHealth(int healthScore){
26           if (!isAlive()) return; // 1.9 - не жив значит прерываем метод.
27           this.healthScore =
28                   Math.min(this.healthScore + healthScore, maxHealthScore);
29           // 1.10 -  FIXME МЕетод - Math.min - объяснение присваиваем персонажу текущее здоровье this.healthScore + переданое в аргумент healthScore и если оно больше максимального maxHealthScore, то метод Math.min - возвращает maxHealthScore.
30       }
31       // 1.6 - уменьшить здоровье на healthScore, если персонаж жив, если не жив - !isAlive(), то прерываем метод.
32       public void minusHealth(int healthScore){
33           if (!isAlive()) return;
34           this.healthScore -= healthScore;
35       }
36       //  1.5 - проверка жив или нет. Метод вернет true, если здоровье больше 0, false, если нет
37       public boolean isAlive() {
38           return healthScore > 0;
39       }
40   
41       // 2.11 - FIXME "АБСТРАКТНЫЙ метод". У всех юнитов должен быть метод rest но реализован он должен быть по разному. Для этого неодходимо исполтзовать АБСТРАКТНЫЙ метод, абстрактный метод - метод без реализации (без {}) -> King 2.12
42       public abstract void rest();
