34           if (gold < 250) {
35               System.out.println("Стоимость армии 250. У короля " + gold);
36               return;

41       public void changeUnits(){   // 2.40 - FIXME  заменяет погибших юнитов
42           /*if (gold < 20) { 
43               System.out.println("Стоимость одного юнита 20. У короля" + gold); 
44               return; 
45           }*/
46           for (int i = 0; i < army.length; i++) {
47               if (gold < 20) { // 2.40 - проверяем есть ли золото хотя бы на одного юнита
48                   System.out.println("Стоимость одного юнита 20. У короля" + gold);
49                   return;
50               }
51               // 2.41 - идем по армии, элемент масива не null и он не жив, заменяем на юнита.
52               if (army[i] != null && !army[i].isAlive()) {
53                   gold -= 20;
54                   army[i] = BattleUnit.getBattleUnit();
55               }
56           }
57       }

59       public void startBattle(King enemy){  // 2.41 - метод битв армий короля
60           // iter
61           /* for (BattleUnit battleUnit : army) { } */
62           for (int i = 0; i < army.length; i++) {
63               int unitIndex = (int) (Math.random() * army.length);
64               int enemyUnitIndex = (int) (Math.random() * enemy.army.length);
65               army[unitIndex].attack(enemy.army[enemyUnitIndex]); // 2.42 - атака рандомного юнита на юнит из армии врага, затем отдых
66               army[unitIndex].rest();
67               enemy.army[enemyUnitIndex].rest();
68           }
69       }
70   
71       // 2.43 - проверяет количество живых юнитов в армии возвращает количество - return count; (ВСЕ!)
72       public int getBattleUnitsCount() {
73           int count = 0;
74           for (BattleUnit battleUnit : army) {
75               if (battleUnit.isAlive()) count += 1;
76           }
77           return count;
78       }
